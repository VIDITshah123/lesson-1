Conditional Statements

function checkAge() {
    const age = document.getElementById('age-input').value;
    if (age >= 18) {
        document.getElementById('age-output').textContent = 'You are an adult.';
    } else {
        document.getElementById('age-output').textContent = 'You are a minor.';
    }
}
